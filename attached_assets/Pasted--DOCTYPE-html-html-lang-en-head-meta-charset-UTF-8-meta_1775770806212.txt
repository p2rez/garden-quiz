let state = { screen:"landing", name:"", answers:[], current:0, result:null, adminPass:"", adminUnlocked:false, submissions:[] };

function render() {
  const app = document.getElementById("app");
  if (state.screen === "landing") app.innerHTML = landingHTML();
  else if (state.screen === "quiz") app.innerHTML = quizHTML();
  else if (state.screen === "loading") app.innerHTML = loadingHTML();
  else if (state.screen === "result") app.innerHTML = resultHTML();
  else if (state.screen === "admin") app.innerHTML = adminHTML();
  bindEvents();
}

function landingHTML() {
  return `
    <h1>What color are you in my garden?</h1>
    <p class="sub">10 questions. One color. Your Texas native plant match.</p>
    <div class="card">
      <label style="font-size:13px;color:#888;display:block;margin-bottom:6px">Your name</label>
      <input type="text" id="nameInput" placeholder="Enter your name" maxlength="50" value="${state.name}" />
      <p class="notice">Your name and answers are saved for review by the quiz creator only.</p>
      <button class="btn-primary" id="startBtn" style="margin-top:1rem" ${!state.name.trim() ? "disabled" : ""}>Start the quiz →</button>
    </div>
    <p class="admin-link" id="adminLink">Admin access</p>`;
}

function quizHTML() {
  const q = questions[state.current];
  const pct = Math.round((state.current / questions.length) * 100);
  return `
    <div class="progress-label"><span>Question ${state.current+1} of ${questions.length}</span><span>${pct}%</span></div>
    <div class="progress-bar"><div class="progress-fill" style="width:${pct}%"></div></div>
    <div class="card">
      <p class="question-text">${q.text}</p>
      ${q.opts.map(o => `<button class="opt-btn" data-opt="${o.id}"><span class="opt-letter">${o.id}.</span>${o.text}</button>`).join("")}
      ${state.current > 0 ? `<button class="btn-outline" id="backBtn" style="margin-top:4px">← Back</button>` : ""}
    </div>`;
}

function loadingHTML() {
  return `<div class="loading-wrap"><div class="spinner"></div><p style="color:#888;font-size:15px">Finding your garden color…</p></div>`;
}

function bindEvents() {
  const on = (id, ev, fn) => { const el = document.getElementById(id); if (el) el.addEventListener(ev, fn); };

  on("nameInput","input", e => { state.name = e.target.value; const btn = document.getElementById("startBtn"); if(btn) btn.disabled = !state.name.trim(); });
  on("nameInput","keydown", e => { if(e.key==="Enter" && state.name.trim()) startQuiz(); });
  on("startBtn","click", startQuiz);
  on("adminLink","click", () => { state.screen="admin"; render(); });
  on("backBtn","click", () => { state.current--; render(); });
  on("retakeBtn","click", () => { state.name=""; state.answers=[]; state.current=0; state.result=null; state.screen="landing"; render(); });
  on("adminPassInput","input", e => state.adminPass = e.target.value);
  on("adminPassInput","keydown", e => { if(e.key==="Enter") loginAdmin(); });
  on("adminLoginBtn","click", loginAdmin);
  on("backToLandingBtn","click", () => { state.screen="landing"; render(); });
  on("refreshBtn","click", loadSubmissions);
  on("signOutBtn","click", () => { state.adminUnlocked=false; state.adminPass=""; state.submissions=[]; state.screen="landing"; render(); });

  document.querySelectorAll(".opt-btn").forEach(btn => {
    btn.addEventListener("click", () => selectAnswer(btn.dataset.opt));
  });
  document.querySelectorAll(".submission-header").forEach(el => {
    el.addEventListener("click", () => {
      const i = parseInt(el.dataset.row);
      state.expandedRow = state.expandedRow === i ? null : i;
      render();
    });
  });
}

function startQuiz() { if(state.name.trim()) { state.screen="quiz"; state.current=0; state.answers=[]; render(); } }

function selectAnswer(optId) {
  const q = questions[state.current];
  state.answers[state.current] = { option: optId, text: q.opts.find(o=>o.id===optId)?.text||"", question: q.text };
  if (state.current < questions.length - 1) { state.current++; render(); }
  else submitQuiz();
}

async function submitQuiz() {
  state.screen = "loading"; render();
  try {
    const res = await fetch("/api/submit", {
      method:"POST", headers:{"Content-Type":"application/json"},
      body: JSON.stringify({ name: state.name.trim(), answers: state.answers })
    });
    const data = await res.json();
    if (data.error) throw new Error(data.error);
    state.result = { color: data.color, reason: data.reason };
  } catch(e) {
    state.result = { color:"Coral", reason:"Something went beautifully wrong — but you're still coral." };
  }
  state.screen = "result"; render();
}

async function loginAdmin() {
  const res = await fetch("/api/admin/login", {
    method:"POST", headers:{"Content-Type":"application/json"},
    body: JSON.stringify({ password: state.adminPass })
  });
  const data = await res.json();
  if (data.success) { state.adminUnlocked=true; state.adminError=""; loadSubmissions(); }
  else { state.adminError = data.error || "Incorrect password."; render(); }
}

async function loadSubmissions() {
  const res = await fetch("/api/admin/submissions", {
    method:"POST", headers:{"Content-Type":"application/json"},
    body: JSON.stringify({ password: state.adminPass })
  });
  const data = await res.json();
  state.submissions = data.submissions || [];
  render();
}

render();
